00252     {
00253         length_type length;
00254         int slot = _hash(str, length);
00255         char *p = _data[slot];
00256         if (p) {
00257             size_type occupied;
00258             if (_search(str, p, length, occupied) != NULL) {
00259                 // str is already in the table. Nothing needs to be done.
00260                 return false;

00264             size_type current = *((size_type *) (p));
00265             size_type required = occupied + sizeof(length_type) + length;
00266             if (required > current) {
00267                 _grow_slot(slot, current, required);

00715     int _hash(const char *str, length_type &length, int seed = 23) const
00716     {
00717         int h = seed;
00718         length = 0;
00719         while (str[length]) {
00720             // Hash this character.
00721             h = h ^ ((h << 5) + (h >> 2) + str[length]);
00722             ++length;
00723         }
00724 
00725         ++length; // include space for the NULL terminator
00726         return h & (_traits.slot_count - 1); // same as h %
00727                                              // _traits.slot_count if
00728                                              // _traits.slot_count is a
00729                                              // power of 2
00730     }

00745     char *_search(const char *str, char *p, length_type length,
00746             size_type &occupied) const
00747     {
00748         occupied = -1;
00749         char *start = p;
00750 
00751         // Search for str in the slot p points to.
00752         p += sizeof(size_type); // skip past size at beginning of slot
00753         length_type w = *((length_type *) p);
00754         while (w != 0) {
00755             p += sizeof(length_type);
00756             if (w == length) {
00757                 // The string being scanned is the same length as str.
00758                 // Make sure they aren't the same string.
00759                 if (strncmp(str, p, length) == 0) {
00760                     // Found str.
00761                     return p - sizeof(length_type);
00762                 }
00763             }
00764             p += w;
00765             w = *((length_type *) p);
00766         }
00767         occupied = p - start + sizeof(length_type);
00768         return NULL;

00811     {

00830     {
00831         int length = *(length_type *) (p);
00832         size_type size = *((size_type *) _data[slot]);
