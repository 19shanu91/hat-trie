00406     int _hash(const char *str, length_type &length, int seed = 23) const {
00407         int h = seed;
00408         length = 0;
00409         while (str[length]) {
00410             // Hash this character.
00411             h = h ^ ((h << 5) + (h >> 2) + str[length]);
00412             ++length;
00413         }
00414 
00415         ++length;  // include space for the NULL terminator
00416         return h & (_traits.slot_count - 1);  // same as h %
00417                                               // _traits.slot_count if
00418                                               // _traits.slot_count is a
00419                                               // power of 2
00420     }
