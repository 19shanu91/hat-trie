00046 inline int ht_alphanumeric_index(char ch) {
00047     if (ch >= '0' && ch <= '9') {
00048         return ch - '0';
00049     } else if (ch >= 'A' && ch <= 'Z') {
00050         return ch - 'A' + 10;
00051     }
00052     return ch - 'a' + 36;
00053 }

00055 template <int alphabet_size, int (*indexof)(char)>
00056 int ht_get_index(char ch) throw(unindexed_character) {
00057     int result = indexof(ch);
00058     if (result < 0 || result >= alphabet_size) {
00059         throw unindexed_character();
00060     }
00061     return result;
00062 }
