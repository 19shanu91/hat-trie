00419     bool insert(const char *word) {
00420         const char *pos = word;
00421         _node_pointer n = _locate(pos);
00422         if (*pos == '\0') {
00423             // word was found in the trie's structure. Mark its location
00424             // as the end of a word.
00425             if (n.p->word() == false) {
00426                 n.p->set_word(true);
00427                 ++_size;
00428                 return true;
00429             }
00430             return false;
00431 
00432         } else {
00433             // word was not found in the trie's structure. Either make a
00434             // new container for it or insert it into an already
00435             // existing container.
00436             _container *c = NULL;
00437             if (n.type == NODE_POINTER) {
00438                 // Make a new container for word.
00439                 _node *p = (_node *) n.p;
00440                 int index = *pos;
00441                 c = new _container(index, _ah_traits);
00442 
00443                 // Insert the new container into the trie structure.
00444                 c->_parent = p;
00445                 p->_children[index] = c;
00446                 p->_types[index] = CONTAINER_POINTER;
00447                 ++pos;
00448             } else if (n.type == CONTAINER_POINTER) {
00449                 // The container for s already exists.
00450                 c = (_container *) n.p;
00451             }
00452 
00453             // Insert the rest of word into the container.
00454             return _insert(c, pos);
00455         }
00456     }

00804     _node_pointer _locate(const char *&s) const {
00805         _node *p = _root;
00806         _node_base *v = NULL;
00807         while (*s) {
00808             int index = *s;
00809             v = p->_children[index];
00810             if (v) {
00811                 ++s;
00812                 if (p->_types[index] == NODE_POINTER) {
00813                     // Keep moving down the trie structure.
00814                     p = (_node *) v;
00815                 } else if (p->_types[index] == CONTAINER_POINTER) {
00816                     // s should appear in the container v
00817                     return _node_pointer(CONTAINER_POINTER, v);

00819             } else {
00820                 // s should appear underneath this node

00826         // s, meaning node n represents s in the trie.

00884     void _burst(_container *htc) {
00885         // Construct a new node.
00886         _node *result = new _node(htc->ch());
00887         result->set_word(htc->word());
00888 
00889         // Make a set of containers for the data in the old container and
00890         // add them to the new node.
00891         typename array_hash<key_type>::iterator it;
00892         for (it = htc->_store.begin(); it != htc->_store.end(); ++it) {
00893             int index = (*it)[0];
00894 
00895             // Do we need to make a new container?
00896             if (result->_children[index] == NULL) {
00897                 // Make a new container and position it under the new node.
00898                 _container *insertion = new _container((*it)[0], _ah_traits);
00899                 insertion->_parent = result;
00900                 result->_children[index] = insertion;
00901                 result->_types[index] = CONTAINER_POINTER;
00902 
00903                 // Set the new container's word field.
00904                 insertion->set_word(((*it)[1] == '\0'));
00905             }
00906 
00907             // Insert the rest of the word into a container.
00908             if ((*it)[1] != '\0') {
00909                 // Insert the rest of the word into the right container.
00910                 ((_container *) result->_children[index])->insert((*it) + 1);
00911             } else {
00912                 // Mark this container as a word.
00913                 ((_container *) result->_children[index])->set_word(true);
00914             }
00915         }
00916 
00917         // Position the new node in the trie.
00918         _node *p = htc->_parent;
00919         result->_parent = p;
00920         int index = htc->ch();
00921         p->_children[index] = result;
00922         p->_types[index] = NODE_POINTER;
00923         delete htc;
00924     }

00935     static _node_pointer _next_child(_node *p, size_type pos, key_type &word) {
00936         _node_pointer result;
00937 
00938         // Search for the next child under this node starting at pos.
00939         for (int i = pos; i < HT_ALPHABET_SIZE && result.p == NULL; ++i) {
00940             if (p->_children[i]) {
00941                 // Found a child.
00942                 result.p = p->_children[i];
00943                 result.type = p->_types[i];
00944 
00945                 // Add this motion to the word.
00946                 word += result.p->ch();
00947             }
00948         }
00949         return result;
00950     }

00962     static _node_pointer _next_word(_node_pointer n, key_type &word) {
00963         // Stop early if we get a NULL pointer.
00964         if (n.p == NULL) { return _node_pointer(); }
00965 
00966         _node_pointer result;
00967         if (n.type == NODE_POINTER) {
00968             // Move to the leftmost child under this node.
00969             result = _next_child((_node *) n.p, 0, word);
00970         }
00971 
00972         if (result.p == NULL) {
00973             // This node has no children. Move up in the trie until
00974             // we can move right.
00975             _node_pointer next;
00976             int pos;
00977             while (n.p->_parent && next.p == NULL) {
00978                 // Looks like we can't move to the right. Move up a level
00979                 // in the trie and try again.
00980                 pos = _pop_back(word) + 1;
00981                 next = _next_child(n.p->_parent, pos, word);
00982                 n = n.p->_parent;
00983             }
00984             result = next;
00985         }
00986 
00987         // Return the lexicographically least node underneath this one.
00988         return _least(result, word);
00989     }

00999     static _node_pointer _least(_node_pointer n, key_type &word) {
01000         while (n.p && n.p->word() == false && n.type == NODE_POINTER) {
01001             // Find the leftmost child of this node and move in
01002             // that direction.
01003             n = _next_child((_node *) n.p, 0, word);
01004         }
01005         return n;
01006     }

01017     static int _pop_back(key_type &word) {
01018         int result = word[word.size() - 1];
01019         word.erase(word.size() - 1);
01020         return result;
01021     }
