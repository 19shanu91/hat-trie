00419     bool insert(const char *word) {
00420         const char *pos = word;
00421         _node_pointer n = _locate(pos);
00422         if (*pos == '\0') {
00423             // word was found in the trie's structure. Mark its location
00424             // as the end of a word.
00425             if (n.p->word() == false) {
00426                 n.p->set_word(true);
00427                 ++_size;
00428                 return true;
00429             }
00430 
00431             // word was already in the trie
00432             return false;
00433 
00434         } else {
00435             // word was not found in the trie's structure. Either make a
00436             // new container for it or insert it into an already
00437             // existing container.
00438             _container *c = NULL;
00439             if (n.type == NODE_POINTER) {
00440                 // Make a new container for word.
00441                 _node *p = (_node *) n.p;
00442                 int index = *pos;
00443                 c = new _container(index, _ah_traits);
00444 
00445                 // Insert the new container into the trie's structure.
00446                 c->_parent = p;
00447                 p->_children[index] = c;
00448                 p->_types[index] = CONTAINER_POINTER;
00449                 ++pos;
00450             } else if (n.type == CONTAINER_POINTER) {
00451                 // The container for s already exists.
00452                 c = (_container *) n.p;
00453             }
00454 
00455             // Insert the rest of word into the container.
00456             return _insert(c, pos);
00457         }
00458     }

00502         if (pos._position.type == CONTAINER_POINTER) {
00503             _container *c = (_container *)pos._position.p;
00504             if (pos._word) {

00541     size_type erase(const key_type &key) {
00542         const char *ps = ref(key).c_str();
00543         _node_pointer n = _locate(ps);
00544         _node *current = NULL;
00545         int result = 0;
00546 
00547         if (n.type == CONTAINER_POINTER) {
00548             // The word is either in a container or is represented by the
00549             // container itself.
00550             _container *c = (_container *)n.p;
00551             result = c->erase(ps);
00552             if (result > 0 && c->size() == 0 && c->word() == false) {
00553                 // Erase the container.
00554                 current = c->_parent;
00555                 delete c;
00556 
00557                 // Mark the container's slot in its parent's _children
00558                 // array as NULL.
00559                 for (int i = 0; i < HT_ALPHABET_SIZE; ++i) {
00560                     if (current->_children[i] == c) {
00561                         current->_children[i] = NULL;
00562                         break;
00563                     }
00564                 }
00565             }
00566 
00567         } else {
00568             // The word is represented by a node in the trie. Set the word
00569             // field on the node to false.

00571             current->set_word(false);

00632     iterator find(const key_type &key) const {
00633         const std::string &word = ref(key);
00634         const char *ps = word.c_str();
00635         _node_pointer n = _locate(ps);
00636 
00637         iterator result;
00638         if (*ps == '\0') {
00639             // The word is in the trie at the node returned by _locate
00640             if (n.p->word()) {
00641                 result = n;
00642                 result._cached_word = std::string(word.c_str());
00643             } else {
00644                 // The word is not a word in the trie
00645                 result = end();
00646             }
00647         } else {
00648             if (n.type == CONTAINER_POINTER) {
00649                 // The word could be in this container
00650                 _container *c = (_container *)n.p;
00651                 array_hash<std::string>::iterator it = c->_store.find(ps);
00652                 if (it != c->_store.end()) {
00653                     // The word is in the trie
00654                     result._position = n;
00655                     result._word = false;
00656                     result._cached_word = std::string(word.c_str(), ps);
00657                     result._container_iterator = it;
00658                 } else {
00659                     // The word is not in the trie
00660                     result = end();
00661                 }
00662             } else {
00663                 // The word is not in the trie
00664                 result = end();
00665             }
00666         }
00667         return result;
00668     }

00923     _node_pointer _locate(const char *&s) const {
00924         _node *p = _root;
00925         _node_base *v = NULL;
00926         while (*s) {
00927             int index = *s;
00928             v = p->_children[index];
00929             if (v) {
00930                 ++s;
00931                 if (p->_types[index] == NODE_POINTER) {
00932                     // Keep moving down the trie structure.
00933                     p = (_node *) v;
00934                 } else if (p->_types[index] == CONTAINER_POINTER) {
00935                     // s should appear in the container v
00936                     return _node_pointer(CONTAINER_POINTER, v);
00937                 }
00938             } else {
00939                 // s should appear underneath this node
00940                 return _node_pointer(NODE_POINTER, p);
00941             }
00942         }
00943 
00944         // If we get here, no container was found that could have held
00945         // s, meaning node n represents s in the trie.
00946         return _node_pointer(NODE_POINTER, p);
00947     }

00984     void _erase_empty_nodes(_node *current) {
00985         while (current && current != _root && current->word() == false) {
00986             // Erase all the nodes that aren't words and don't
00987             // have any children above the erased node or container.
00988             // Start by determining whether the current node has any
00989             // children.
00990             bool children = false;
00991             for (int i = 0; i < HT_ALPHABET_SIZE && !children; ++i) {
00992                 children |= (bool)current->_children[i];
00993             }
00994 
00995             // If the current node doesn't have any children and isn't a
00996             // word, delete it.
00997             if (children == false) {
00998                 _node *tmp = current;
00999                 current = current->_parent;
01000                 delete tmp;

01119     static _node_pointer _next_word(_node_pointer n, key_type &word) {
01120         // Stop early if we get a NULL pointer.
01121         if (n.p == NULL) { return _node_pointer(); }
01122 
01123         _node_pointer result;
01124         if (n.type == NODE_POINTER) {
01125             // Move to the leftmost child under this node.
01126             result = _next_child((_node *) n.p, 0, word);
01127         }
01128 
01129         if (result.p == NULL) {
01130             // This node has no children. Move up in the trie until
01131             // we can move right.
01132             _node_pointer next;
01133             int pos;
01134             while (n.p->_parent && next.p == NULL) {
01135                 // Looks like we can't move to the right. Move up a level
01136                 // in the trie and try again.
01137                 pos = _pop_back(word) + 1;
01138                 next = _next_child(n.p->_parent, pos, word);
01139                 n = n.p->_parent;
01140             }
01141             result = next;
01142         }
01143 
01144         // Return the lexicographically least node underneath this one.
01145         return _least(result, word);

01157         while (n.p && n.p->word() == false && n.type == NODE_POINTER) {
