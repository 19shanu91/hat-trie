00421     bool insert(const char *word) {
00422         const char *pos = word;
00423         htnode_ptr n = _locate(pos);
00424         if (*pos == '\0') {
00425             // word was found in the trie's structure. Mark its location
00426             // as the end of a word.
00427             if (n.word() == false) {
00428                 n.set_word(true);
00429                 ++_size;
00430                 return true;
00431             }
00432 
00433             // word was already in the trie
00434             return false;
00435 
00436         } else {
00437             // word was not found in the trie's structure. Either make a
00438             // new bucket for it or insert it into an already
00439             // existing bucket
00440             ahnode *at = NULL;
00441             if (n.type == NODE_POINTER) {
00442                 // Make a new bucket for word
00443                 htnode *p = n.ptr.node;
00444                 int index = *pos;
00445 
00446                 at = new ahnode();
00447                 at->table = new bucket(_ah_traits);
00448                 at->ch = index;
00449                 at->word = false;
00450 
00451                 // Insert the new bucket into the trie's structure
00452                 at->parent = p;
00453                 p->children[index].bucket = at;
00454                 p->types[index] = BUCKET_POINTER;
00455                 ++pos;
00456             } else if (n.type == BUCKET_POINTER) {
00457                 // The container for s already exists.
00458                 at = n.ptr.bucket;
00459             }
00460 
00461             // Insert the rest of word into the container.
00462             return _insert(at, pos);
00463         }
00464     }

00551         htnode *current = NULL;
00552         int result = 0;
00553 
00554         if (n.type == BUCKET_POINTER) {
00555             // The word is either in a container or is represented by the

00640     iterator find(const key_type &key) const {
00641         const std::string &word = ref(key);
00642         const char *ps = word.c_str();
00643         htnode_ptr n = _locate(ps);
00644 
00645         iterator result;
00646         if (*ps == '\0') {
00647             // The word is in the trie at the node returned by _locate
00648             if (n.word()) {
00649                 result = n;
00650                 result._cached_word = std::string(word.c_str());
00651             } else {
00652                 // The word is not a word in the trie
00653                 result = end();
00654             }
00655         } else {
00656             if (n.type == BUCKET_POINTER) {
00657                 // The word could be in this container
00658                 ahnode *b = n.ptr.bucket;
00659                 bucket::iterator it = b->table->find(ps);
00660                 if (it != b->table->end()) {
00661                     // The word is in the trie
00662                     result._position = n;
00663                     result._word = false;
00664                     result._cached_word = std::string(word.c_str(), ps);
00665                     result._container_iterator = it;
00666                 } else {
00667                     // The word is not in the trie
00668                     result = end();
00669                 }
00670             } else {
00671                 // The word is not in the trie
00672                 result = end();
00673             }
00674         }
00675         return result;

00929     htnode_ptr _locate(const char *&s) const {
00930         htnode *p = _root;
00931         child_ptr v;
00932         while (*s) {
00933             int index = *s;
00934             v = p->children[index];
00935             if (v.bucket) {
00936                 ++s;
00937                 if (p->types[index] == NODE_POINTER) {
00938                     // Keep moving down the trie structure.
00939                     p = v.node;
00940                 } else if (p->types[index] == BUCKET_POINTER) {
00941                     // s should appear in the container v
00942                     return htnode_ptr(v, BUCKET_POINTER);
00943                 }
00944             } else {
00945                 // s should appear underneath this node
00946                 return htnode_ptr(p);
00947             }
00948         }
00949 
00950         // If we get here, no container was found that could have held
00951         // s, meaning node n represents s in the trie.
00952         return htnode_ptr(p);
00953     }

00968     bool _insert(ahnode *htc, const char *s) {
00969         // Try to insert s into the container.
00970         bool result;
00971         if (*s == '\0') {
00972             result = !htc->word;
00973             htc->word = true;
00974         } else {
00975             result = htc->table->insert(s);
00976         }
00977 
00978         if (result) {
00979             ++_size;
00980             if (_traits.burst_threshold > 0 &&
00981                     htc->table->size() > _traits.burst_threshold) {
00982                 // burst the bucket into nodes
00983                 _burst(htc);
00984             }
00985             return true;
00986         }
00987         return false;
00988     }

00998     void _erase_empty_nodes(htnode *current) {
00999         while (current && current != _root && current->word() == false) {
01000             // Erase all the nodes that aren't words and don't
01001             // have any children above the erased node or container.
01002             // Start by determining whether the current node has any
01003             // children.
01004             bool children = false;
01005             for (int i = 0; i < HT_ALPHABET_SIZE && !children; ++i) {
01006                 children |= (bool)current->children[i].bucket;
01007             }
01008 
01009             // If the current node doesn't have any children and isn't a
01010             // word, delete it.
01011             if (children == false) {
01012                 htnode *tmp = current;
01013                 current = current->parent;
01014                 delete tmp;
01015 
01016                 // Mark the slot in current's parent's children array
01017                 // as NULL.
01018                 for (int i = 0; i < HT_ALPHABET_SIZE; ++i) {
01019                     if (current->children[i].node == tmp) {
01020                         current->children[i].node = NULL;
01021                         break;
01022                     }
01023                 }
01024             } else {
01025                 // Stop the while loop.
01026                 current = NULL;
01027             }
01028         }
01029     }

01137     static htnode_ptr _next_word(htnode_ptr n, key_type &word) {
01138         // Stop early if we get a NULL pointer.
01139         if (n.ptr.node == NULL) { return htnode_ptr(); }
01140 
01141         htnode_ptr result;
01142         if (n.type == NODE_POINTER) {
01143             // Move to the leftmost child under this node.
01144             result = _next_child(n.ptr.node, 0, word);
01145         }

01147         if (result.ptr.node == NULL) {
01148             // This node has no children. Move up in the trie until
01149             // we can move right.
01150             htnode_ptr next;
01151             int pos;

01154                 // Looks like we can't move to the right. Move up a level
01155                 // in the trie and try again.
01156                 pos = _pop_back(word) + 1;

01159                 parent = n.ptr.node->parent;
