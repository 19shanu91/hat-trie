00386     bool insert(const char *word) {
00387         const char *pos = word;
00388         _node_pointer n = _locate(pos);
00389         if (*pos == '\0') {
00390             // word was found in the trie's structure. Mark its location
00391             // as the end of a word.
00392             if (n.p->word() == false) {
00393                 n.p->set_word(true);
00394                 ++_size;
00395                 return true;
00396             }
00397             return false;
00398 
00399         } else {
00400             // word was not found in the trie's structure. Either make a
00401             // new container for it or insert it into an already
00402             // existing container.
00403             _container *c = NULL;
00404             if (n.type == NODE_POINTER) {
00405                 // Make a new container for word.
00406                 _node *p = (_node *) n.p;
00407                 int index = *pos;
00408                 c = new _container(index, _ah_traits);
00409 
00410                 // Insert the new container into the trie structure.
00411                 c->_parent = p;
00412                 p->_children[index] = c;
00413                 p->_types[index] = CONTAINER_POINTER;
00414                 ++pos;
00415             } else if (n.type == CONTAINER_POINTER) {
00416                 // The container for s already exists.
00417                 c = (_container *) n.p;
00418             }
00419 
00420             // Insert the rest of word into the container.
00421             return _insert(c, pos);
00422         }
00423     }

00771     _node_pointer _locate(const char *&s) const {
00772         _node *p = _root;
00773         _node_base *v = NULL;
00774         while (*s) {
00775             int index = *s;
00776             v = p->_children[index];
00777             if (v) {
00778                 ++s;
00779                 if (p->_types[index] == NODE_POINTER) {
00780                     // Keep moving down the trie structure.
00781                     p = (_node *) v;
00782                 } else if (p->_types[index] == CONTAINER_POINTER) {
00783                     // s should appear in the container v
00784                     return _node_pointer(CONTAINER_POINTER, v);
00785                 }
00786             } else {
00787                 // s should appear underneath this node
00788                 return _node_pointer(NODE_POINTER, p);
00789             }
00790         }
00791 
00792         // If we get here, no container was found that could have held
00793         // s, meaning node n represents s in the trie.
00794         return _node_pointer(NODE_POINTER, p);
00795     }

00810     bool _insert(_container *htc, const char *s) {
00811         // Try to insert s into the container.

00851     void _burst(_container *htc) {
00852         // Construct a new node.
00853         _node *result = new _node(htc->ch());
00854         result->set_word(htc->word());
00855 
00856         // Make a set of containers for the data in the old container and
00857         // add them to the new node.
00858         typename array_hash<key_type>::iterator it;
00859         for (it = htc->_store.begin(); it != htc->_store.end(); ++it) {
00860             int index = (*it)[0];
00861 
00862             // Do we need to make a new container?
00863             if (result->_children[index] == NULL) {
00864                 // Make a new container and position it under the new node.
00865                 _container *insertion = new _container((*it)[0], _ah_traits);
00866                 insertion->_parent = result;
00867                 result->_children[index] = insertion;
00868                 result->_types[index] = CONTAINER_POINTER;
00869 
00870                 // Set the new container's word field.
00871                 insertion->set_word(((*it)[1] == '\0'));
00872             }
00873 
00874             // Insert the rest of the word into a container.
00875             if ((*it)[1] != '\0') {
00876                 // Insert the rest of the word into the right container.
00877                 ((_container *) result->_children[index])->insert((*it) + 1);
00878             } else {
00879                 // Mark this container as a word.
00880                 ((_container *) result->_children[index])->set_word(true);
00881             }
00882         }
00883 
00884         // Position the new node in the trie.
00885         _node *p = htc->_parent;
00886         result->_parent = p;
00887         int index = htc->ch();
00888         p->_children[index] = result;
00889         p->_types[index] = NODE_POINTER;
00890         delete htc;
00891     }

00902     static _node_pointer _next_child(_node *p, size_type pos, key_type &word) {
00903         _node_pointer result;
00904 
00905         // Search for the next child under this node starting at pos.
00906         for (int i = pos; i < HT_ALPHABET_SIZE && result.p == NULL; ++i) {
00907             if (p->_children[i]) {
00908                 // Found a child.
00909                 result.p = p->_children[i];
00910                 result.type = p->_types[i];
00911 
00912                 // Add this motion to the word.
00913                 word += result.p->ch();
00914             }
00915         }
00916         return result;
00917     }

00929     static _node_pointer _next_word(_node_pointer n, key_type &word) {
00930         // Stop early if we get a NULL pointer.
00931         if (n.p == NULL) { return _node_pointer(); }
00932 
00933         _node_pointer result;
00934         if (n.type == NODE_POINTER) {
00935             // Move to the leftmost child under this node.
00936             result = _next_child((_node *) n.p, 0, word);
00937         }
00938 
00939         if (result.p == NULL) {
00940             // This node has no children. Move up in the trie until
00941             // we can move right.
00942             _node_pointer next;
00943             int pos;
00944             while (n.p->_parent && next.p == NULL) {
00945                 // Looks like we can't move to the right. Move up a level
00946                 // in the trie and try again.
00947                 pos = _pop_back(word) + 1;
00948                 next = _next_child(n.p->_parent, pos, word);
00949                 n = n.p->_parent;
00950             }
00951             result = next;
00952         }
00953 
00954         // Return the lexicographically least node underneath this one.
00955         return _least(result, word);
00956     }

00966     static _node_pointer _least(_node_pointer n, key_type &word) {
00967         while (n.p && n.p->word() == false && n.type == NODE_POINTER) {
00968             // Find the leftmost child of this node and move in
00969             // that direction.
00970             n = _next_child((_node *) n.p, 0, word);
00971         }
00972         return n;
00973     }
