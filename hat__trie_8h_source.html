00422     bool insert(const char *word) {
00423         const char *pos = word;
00424         _node_pointer n = _locate(pos);
00425         if (*pos == '\0') {
00426             // word was found in the trie's structure. Mark its location
00427             // as the end of a word.
00428             if (n.p->word() == false) {
00429                 n.p->set_word(true);
00430                 ++_size;
00431                 return true;
00432             }
00433 
00434             // word was already in the trie
00435             return false;
00436 
00437         } else {
00438             // word was not found in the trie's structure. Either make a
00439             // new container for it or insert it into an already
00440             // existing container.
00441             _container *c = NULL;
00442             if (n.type == NODE_POINTER) {
00443                 // Make a new container for word.
00444                 _node *p = (_node *) n.p;
00445                 int index = *pos;
00446                 c = new _container(index, _ah_traits);
00447 
00448                 // Insert the new container into the trie's structure.
00449                 c->_parent = p;
00450                 p->_children[index] = c;
00451                 p->_types[index] = CONTAINER_POINTER;
00452                 ++pos;
00453             } else if (n.type == CONTAINER_POINTER) {
00454                 // The container for s already exists.
00455                 c = (_container *) n.p;
00456             }
00457 
00458             // Insert the rest of word into the container.
00459             return _insert(c, pos);
00460         }
00461     }

00547         _node *current = NULL;

00549 
00550         if (n.type == CONTAINER_POINTER) {
00551             // The word is either in a container or is represented by the

00635     iterator find(const key_type &key) const {
00636         const std::string &word = ref(key);
00637         const char *ps = word.c_str();
00638         _node_pointer n = _locate(ps);
00639 
00640         iterator result;
00641         if (*ps == '\0') {
00642             // The word is in the trie at the node returned by _locate
00643             if (n.p->word()) {
00644                 result = n;
00645                 result._cached_word = std::string(word.c_str());
00646             } else {
00647                 // The word is not a word in the trie
00648                 result = end();
00649             }
00650         } else {
00651             if (n.type == CONTAINER_POINTER) {
00652                 // The word could be in this container
00653                 _container *c = (_container *)n.p;
00654                 array_hash<std::string>::iterator it = c->_store.find(ps);
00655                 if (it != c->_store.end()) {
00656                     // The word is in the trie
00657                     result._position = n;
00658                     result._word = false;

00660                     result._container_iterator = it;
00661                 } else {
00662                     // The word is not in the trie

00926     _node_pointer _locate(const char *&s) const {
00927         _node *p = _root;
00928         _node_base *v = NULL;
00929         while (*s) {
00930             int index = *s;
00931             v = p->_children[index];
00932             if (v) {
00933                 ++s;
00934                 if (p->_types[index] == NODE_POINTER) {
00935                     // Keep moving down the trie structure.
00936                     p = (_node *) v;
00937                 } else if (p->_types[index] == CONTAINER_POINTER) {
00938                     // s should appear in the container v
00939                     return _node_pointer(CONTAINER_POINTER, v);
00940                 }
00941             } else {
00942                 // s should appear underneath this node
00943                 return _node_pointer(NODE_POINTER, p);
00944             }
00945         }
00946 
00947         // If we get here, no container was found that could have held
00948         // s, meaning node n represents s in the trie.
00949         return _node_pointer(NODE_POINTER, p);
00950     }

00965     bool _insert(_container *htc, const char *s) {
00966         // Try to insert s into the container.
00967         if (htc->insert(s)) {
00968             ++_size;
00969             if (_traits.burst_threshold > 0 &&
00970                     htc->size() > _traits.burst_threshold) {
00971                 // The container has too many strings in it; burst the

00973                 _burst(htc);
00974             }
00975             return true;
00976         }
00977         return false;

00988     void _erase_empty_nodes(_node *current) {
00989         while (current && current != _root && current->word() == false) {
00990             // Erase all the nodes that aren't words and don't
00991             // have any children above the erased node or container.
00992             // Start by determining whether the current node has any
00993             // children.
00994             bool children = false;
00995             for (int i = 0; i < HT_ALPHABET_SIZE && !children; ++i) {
00996                 children |= (bool)current->_children[i];
00997             }
00998 
00999             // If the current node doesn't have any children and isn't a
01000             // word, delete it.
01001             if (children == false) {
01002                 _node *tmp = current;
01003                 current = current->_parent;

01096     static _node_pointer _next_child(_node *p, size_type pos, key_type &word) {
01097         _node_pointer result;
01098 
01099         // Search for the next child under this node starting at pos.
01100         for (int i = pos; i < HT_ALPHABET_SIZE && result.p == NULL; ++i) {

01123     static _node_pointer _next_word(_node_pointer n, key_type &word) {
01124         // Stop early if we get a NULL pointer.
01125         if (n.p == NULL) { return _node_pointer(); }
01126 
01127         _node_pointer result;
01128         if (n.type == NODE_POINTER) {
01129             // Move to the leftmost child under this node.
01130             result = _next_child((_node *) n.p, 0, word);
01131         }
01132 
01133         if (result.p == NULL) {
01134             // This node has no children. Move up in the trie until
01135             // we can move right.
01136             _node_pointer next;
01137             int pos;
01138             while (n.p->_parent && next.p == NULL) {
01139                 // Looks like we can't move to the right. Move up a level
01140                 // in the trie and try again.
01141                 pos = _pop_back(word) + 1;
01142                 next = _next_child(n.p->_parent, pos, word);
01143                 n = n.p->_parent;
01144             }
01145             result = next;
01146         }
01147 
01148         // Return the lexicographically least node underneath this one.
01149         return _least(result, word);
01150     }

01160     static _node_pointer _least(_node_pointer n, key_type &word) {
01161         while (n.p && n.p->word() == false && n.type == NODE_POINTER) {
01162             // Find the leftmost child of this node and move in
01163             // that direction.
01164             n = _next_child((_node *) n.p, 0, word);
