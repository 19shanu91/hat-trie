00417     bool insert(const char *word) {
00418         const char *pos = word;
00419         htnode_ptr n = _locate(pos);
00420         if (*pos == '\0') {
00421             // word was found in the trie's structure. Mark its location
00422             // as the end of a word.
00423             if (n.word() == false) {
00424                 n.set_word(true);
00425                 ++_size;
00426                 return true;
00427             }
00428 
00429             // word was already in the trie
00430             return false;
00431 
00432         } else {
00433             // word was not found in the trie's structure. Either make a
00434             // new bucket for it or insert it into an already
00435             // existing bucket
00436             ahnode *at = NULL;
00437             if (n.type == NODE_POINTER) {
00438                 // Make a new bucket for word
00439                 htnode *p = n.ptr.node;
00440                 int index = *pos;
00441 
00442                 at = new ahnode();
00443                 at->table = new bucket(_ah_traits);
00444                 at->ch = index;
00445                 at->word = false;
00446 
00447                 // Insert the new bucket into the trie's structure
00448                 at->parent = p;
00449                 p->children[index].bucket = at;
00450                 p->types[index] = BUCKET_POINTER;
00451                 ++pos;
00452             } else if (n.type == BUCKET_POINTER) {
00453                 // The container for s already exists.
00454                 at = n.ptr.bucket;
00455             }
00456 
00457             // Insert the rest of word into the container.
00458             return _insert(at, pos);
00459         }
00460     }

00636     iterator find(const key_type &key) const {
00637         const std::string &word = ref(key);
00638         const char *ps = word.c_str();
00639         htnode_ptr n = _locate(ps);
00640 
00641         iterator result;
00642         if (*ps == '\0') {
00643             // The word is in the trie at the node returned by _locate
00644             if (n.word()) {
00645                 result = n;
00646                 result._cached_word = std::string(word.c_str());
00647             } else {
00648                 // The word is not a word in the trie
00649                 result = end();
00650             }
00651         } else {
00652             if (n.type == BUCKET_POINTER) {
00653                 // The word could be in this container
00654                 ahnode *b = n.ptr.bucket;
00655                 bucket::iterator it = b->table->find(ps);
00656                 if (it != b->table->end()) {
00657                     // The word is in the trie
00658                     result._position = n;
00659                     result._word = false;
00660                     result._cached_word = std::string(word.c_str(), ps);
00661                     result._container_iterator = it;
00662                 } else {
00663                     // The word is not in the trie
00664                     result = end();
00665                 }
00666             } else {
00667                 // The word is not in the trie

00918     htnode_ptr _locate(const char *&s) const {
00919         htnode *p = _root;
00920         child_ptr v;
00921         while (*s) {
00922             int index = *s;
00923             v = p->children[index];
00924             if (v.bucket) {
00925                 ++s;
00926                 if (p->types[index] == NODE_POINTER) {
00927                     // Keep moving down the trie structure.
00928                     p = v.node;
00929                 } else if (p->types[index] == BUCKET_POINTER) {
00930                     // s should appear in the container v
00931                     return htnode_ptr(v, BUCKET_POINTER);
00932                 }
00933             } else {
00934                 // s should appear underneath this node
00935                 return htnode_ptr(p);
00936             }
00937         }
00938 
00939         // If we get here, no container was found that could have held
00940         // s, meaning node n represents s in the trie.
00941         return htnode_ptr(p);
00942     }

00957     bool _insert(ahnode *htc, const char *s) {
00958         // Try to insert s into the container.
00959         bool result;
00960         if (*s == '\0') {
00961             result = !htc->word;
00962             htc->word = true;
00963         } else {
00964             result = htc->table->insert(s);
00965         }
00966 
00967         if (result) {
00968             ++_size;
00969             if (_traits.burst_threshold > 0 &&
00970                     htc->table->size() > _traits.burst_threshold) {
00971                 // burst the bucket into nodes
00972                 _burst(htc);
00973             }
00974             return true;
00975         }
00976         return false;
00977     }

00987     void _erase_empty_nodes(htnode *current) {
00988         while (current && current != _root && current->word() == false) {
00989             // Erase all the nodes that aren't words and don't
00990             // have any children above the erased node or container.
00991             // Start by determining whether the current node has any
00992             // children.
00993             bool children = false;
00994             for (int i = 0; i < HT_ALPHABET_SIZE && !children; ++i) {
00995                 children |= (bool)current->children[i].bucket;
00996             }
00997 
00998             // If the current node doesn't have any children and isn't a
00999             // word, delete it.
01000             if (children == false) {
01001                 htnode *tmp = current;
01002                 current = current->parent;
01003                 delete tmp;
01004 
01005                 // Mark the slot in current's parent's children array
01006                 // as NULL.
01007                 for (int i = 0; i < HT_ALPHABET_SIZE; ++i) {
01008                     if (current->children[i].node == tmp) {
01009                         current->children[i].node = NULL;
01010                         break;
01011                     }
01012                 }
01013             } else {
01014                 // Stop the while loop.
01015                 current = NULL;
01016             }
01017         }
01018     }

01050     void _burst(ahnode *htc) {
01051         // Construct a new node.
01052         htnode *result = new htnode(htc->ch);
01053         result->set_word(htc->word);
01054 
01055         // Make a set of containers for the data in the old container and
01056         // add them to the new node.
01057         typename bucket::iterator it;
01058         for (it = htc->table->begin(); it != htc->table->end(); ++it) {
01059             int index = (*it)[0];

01126     static htnode_ptr _next_word(htnode_ptr n, key_type &word) {
01127         // Stop early if we get a NULL pointer.
01128         if (n.ptr.node == NULL) { return htnode_ptr(); }
01129 
01130         htnode_ptr result;
01131         if (n.type == NODE_POINTER) {
01132             // Move to the leftmost child under this node.
01133             result = _next_child(n.ptr.node, 0, word);
01134         }
01135 
01136         if (result.ptr.node == NULL) {
01137             // This node has no children. Move up in the trie until
01138             // we can move right.
01139             htnode_ptr next;
01140             int pos;
01141             htnode *parent = n.parent();

01143                 // Looks like we can't move to the right. Move up a level
01144                 // in the trie and try again.
01145                 pos = _pop_back(word) + 1;

01148                 parent = n.ptr.node->parent;
01149             }
